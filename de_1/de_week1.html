Graph Visualization

Code 
# take out the global observations
bp_data = bp_data[~bp_data['Country'].str.contains('World')]

# take out the 'total' column
bp_data = bp_data[~bp_data['source'].str.contains('Total')]


# create graph
bp_data = bp_data.pivot(index='Country', columns='source', values='tpes')
ax = bp_data.plot(kind='bar', stacked=True, figsize=(10, 7))
ax.set_ylabel('TPES (Exajoules)')
ax.set_xlabel('country')
ax.set_title('Primary Energy Consumption by Source')
plt.style.use('ggplot')
plt.xticks(rotation=45)
plt.show()
