import pandas as pd
import seaborn as sns
import requests
from io import BytesIO
from matplotlib import pyplot as plt
from datetime import datetime

DATA_URL = 'https://www.eia.gov/dnav/pet/xls/PET_PRI_SPT_S1_D.xls'
price_data = None

# fetch data from EIA and load it into pandas dataframe
response = requests.get(DATA_URL)
if response.status_code == 200:
    with BytesIO(response.content) as bio:
        price_data = pd.read_excel(
          bio, 
          sheet_name="Data 1",
          skiprows=range(2)
        )
else:
    print(f"Failed to fetch the Excel file. Status code: {response.status_code}")

# rename columns
price_data.columns = ['date', 'wti', 'brent']

# visualize the data
price_data.tail(10)

black = viridis(0.0)  # adjust as needed to get the desired shade
custom_palette = [aqua, black]

# Set figure size
plt.figure(figsize=(14,10))

# Line Plot Config
sns.lineplot(data=price_data, x='date', y='price', hue='crude_stream', palette=custom_palette)

# X-Axis constraints
plt.gca().set_xlim([datetime(2013, 1, 1), datetime(2023, 12, 31)])

# Y-Axis constraints
plt.gca().set_ylim([0, 150])

# Title Styling
plt.title("Crude Oil Spot Prices", loc='left', fontsize=20, weight='bold', pad=20)
plt.text(
  s='Daily West Texas Intermediate (WTI) and Brent prices in dollars per barrel (2013-2023)', 
  x=0,
  y=1.00,
  fontsize=15,
  ha='left',
  transform=plt.gca().transAxes
